\begin{tikzpicture}
    %define coordinate bounds
  \def\xmin{-5}
  \def\xmax{5}
  \def\ymin{-5}
  \def\ymax{5}

  %define coordinates
  \coordinate (origin) at (0,0);

  %draw image bounds
  \fill[white] (\xmin,\ymin)rectangle(\xmax,\ymax);
  %draw axis
  %x-axis
  \draw[line width=0.5pt,white!40!black,>=latex,->]
  (\xmin,0) -- (\xmax,0) node[anchor=north]{$x$};
  %y-axis
  \draw[line width=0.5pt,white!40!black,>=latex,->]
  (0,\ymin) -- (0,\ymax) node[anchor=east]{$y$};

\end{tikzpicture}
